# Gemma 4 torchax trainer — pip deps.
#
# torchax pin matches the wiki submodule
# (`raw/code/torchax`, commit 8f957d1 per `wiki/codebases/torchax.md`).
# The published wheel at this commit is `torchax==0.0.12`; install from
# that commit rather than PyPI to guarantee the op-lowering tables match.
#
# NOTE: `--extra-index-url` and the TPU wheel pin for JAX are libtpu-
# dependent; on a fresh v6e host the canonical recipe is
#   pip install -U "jax[tpu]"
# with Google's libtpu index. We pin a reasonable floor here but let the
# TPU installer pick the libtpu build.

# Torch (CPU wheel). torchax only uses torch for its mode/dispatch
# machinery and pytree; we do NOT want the CUDA build.
--extra-index-url https://download.pytorch.org/whl/cpu
torch>=2.5,<3.0

# JAX. torchax's interop.py uses `jax.shard_map` at the top level, which
# exists from jax 0.8.0. Gemma 4 E4B was released 2026-04; use a recent jax.
jax[tpu]>=0.8.0

# torchax at commit 8f957d1 (== 0.0.12). Install from GitHub to match the
# wiki's pinned submodule commit.
torchax @ git+https://github.com/pytorch/xla.git@8f957d11128ee273a14414750b3998870f933d15#subdirectory=torchax

# Transformers must ship Gemma 4. The config.json we inspected carries
# `transformers_version: 5.5.0.dev0` — Gemma 4 support is on main, not on
# a stable tag yet (as of 2026-04-22). Install from main.
transformers @ git+https://github.com/huggingface/transformers.git

# Dataset loader.
datasets>=3.0

# Optimizer.
optax>=0.2.0

# YAML config support (optional; only needed if you pass --config).
pyyaml>=6.0

# HF auth helpers (model card says Apache-2.0 / not gated but you still
# need to be logged in to hit HF rate-limits cleanly).
huggingface_hub>=0.24
